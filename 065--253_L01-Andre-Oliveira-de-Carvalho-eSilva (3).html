# Funo com apply
com_apply <- function() {
  sapply(mtcars, min)
}

# Funo com for loop
com_for <- function() {
  resultado <- numeric(ncol(mtcars))
  for (i in 1:ncol(mtcars)) {
    resultado[i] <- min(mtcars[[i]])
  }
  names(resultado) <- names(mtcars)
  resultado
}

# Extrair terceira linha usando for loop
extrair_terceira_linha_for <- function(lista_matrizes) {
  resultado <- list()
  
  for (i in 1:length(lista_matrizes)) {
    # i: ndice que percorre cada posio da lista
    # lista_matrizes[[i]]: acessa cada matriz individualmente
    resultado[[i]] <- lista_matrizes[[i]][3, ]  # Extrai a terceira linha
  }
  
  return(resultado)
}

# Executando
resultado_for <- extrair_terceira_linha_for(lista.obj) 

# Extrair terceira linha usando sapply
extrair_terceira_linha_apply <- function(matriz) {
  return(matriz[3, ])
}

resultado_apply <- sapply(X = lista.obj, FUN = extrair_terceira_linha_apply)

10. A funo system.time retorna o tempo de operaes em R. Verifique comoesse comando funciona. Para uma matriz X, 107 2 e um vetor y de comprimento107, calcule (algumas quantas vezes) o produto Xty, usando multiplicao dematrizes e a funo crossproduct. Qual o procedimento mais rpido?

# Criar dados de exemplo
set.seed(123)
n <- 10^7  # 10 milhes de observaes

# Matriz X: 10^7 x 2
X <- matrix(rnorm(2 * n), ncol = 2)

# Vetor y: comprimento 10^7
y <- rnorm(n)

# Mtodo 1: Multiplicao de matrizes tradicional
tempo_tradicional <- system.time({
  resultado_trad <- t(X) %*% y
})

# Mtodo 2: Funo crossprod
tempo_crossprod <- system.time({
  resultado_cross <- crossprod(X, y)
})

# Comparar resultados
print("Tempo - Multiplicao tradicional:")
## [1] "Tempo - Multiplicao tradicional:"
print(tempo_tradicional)
##   usurio   sistema decorrido 
##      0.08      0.03      0.11
print("Tempo - Crossprod:")
## [1] "Tempo - Crossprod:"
print(tempo_crossprod)
##   usurio   sistema decorrido 
##      0.04      0.00      0.03
# Verificar se resultados so iguais
print("Resultados so iguais?")
## [1] "Resultados so iguais?"
print(all.equal(as.numeric(resultado_trad), as.numeric(resultado_cross)))

repeticoes <- 10
tempos_trad <- numeric(repeticoes)
tempos_cross <- numeric(repeticoes)

for(i in 1:repeticoes) {
  tempos_trad[i] <- system.time({ t(X) %*% y })["elapsed"]
  tempos_cross[i] <- system.time({ crossprod(X, y) })["elapsed"]
